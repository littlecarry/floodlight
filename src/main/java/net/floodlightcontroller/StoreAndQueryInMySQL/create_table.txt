

CREATE TABLE
    packet_info
    (
        id INT NOT NULL AUTO_INCREMENT,
        switch_id BIGINT NOT NULL,
        srcAdd BIGINT NOT NULL,
        dstAdd BIGINT NOT NULL,
        srcMac BIGINT NOT NULL,
        dstMac BIGINT NOT NULL,
        byteCount INT NOT NULL,
        protocolType VARCHAR(20) NOT NULL,
        PRIMARY KEY (id)
    )
    ENGINE=InnoDB DEFAULT CHARSET=utf8;
